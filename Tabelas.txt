CREATE TABLE MovimentosFinanceiro (
  id_movto INTEGER PRIMARY KEY,
  descricao_movto TEXT NOT NULL,
  data_movto DATE NOT NULL,
  hora_movto TIMESTAMP,
  usuario_movto CHAR
);

CREATE TABLE Enderecos (
  id_endereco INTEGER PRIMARY KEY,
  endereco CHAR NOT NULL,
  cidade CHAR NOT NULL,
  estado CHAR NOT NULL,
  pais CHAR NOT NULL,
  cep INTEGER NOT NULL
);

CREATE TABLE Categorias (
  id_categoria INTEGER PRIMARY KEY,
  nome_categoria CHAR
);

CREATE TABLE Clientes (
  id_cliente INTEGER PRIMARY KEY,
  id_endereco INTEGER,
  nome_cliente CHAR NOT NULL,
  cnpj_cliente INTEGER NOT NULL,
  telefone INTEGER NOT NULL,
  email_cliente CHAR NOT NULL,
  FOREIGN KEY (id_endereco) REFERENCES Enderecos(id_endereco)
);

CREATE TABLE Fornecedores (
  id_fornecedor INTEGER PRIMARY KEY,
  id_endereco INTEGER,
  nome_fornecedor CHAR NOT NULL,
  cnpj_fornecedor INTEGER NOT NULL,
  telefone_fornecedor INTEGER NOT NULL,
  email_fornecedor CHAR NOT NULL,
  FOREIGN KEY (id_endereco) REFERENCES Enderecos(id_endereco)
);

CREATE TABLE Produtos (
  id_produto INTEGER PRIMARY KEY,
  id_fornecedor INTEGER,
  id_categoria INTEGER,
  nome_prod char NOT NULL,
  valor_prod INTEGER NOT NULL,
  FOREIGN KEY (id_fornecedor) REFERENCES Fornecedores(id_fornecedor),
  FOREIGN KEY (id_categoria) REFERENCES Categorias(id_categoria)
);

CREATE TABLE Saidas (
  id_saida INTEGER PRIMARY KEY,
  id_cliente INTEGER,
  FOREIGN KEY (id_cliente) REFERENCES Clientes(id_cliente)
);

CREATE TABLE Entradas (
  id_entrada INTEGER PRIMARY KEY,
  id_fornecedor INTEGER,
  valor_entrada INTEGER NOT NULL,
  quantidade_entrada INTEGER NOT NULL,
  FOREIGN KEY (id_fornecedor) REFERENCES Fornecedores(id_fornecedor)
);

CREATE TABLE ItemPedido (
  id_pedido INTEGER PRIMARY KEY,
  id_produto INTEGER,
  valor_pedido INTEGER NOT NULL,
  quantidade_pedido INTEGER NOT NULL,
  id_saida INTEGER,
  FOREIGN KEY (id_saida) REFERENCES Saidas(id_saida),
  FOREIGN KEY (id_produto) REFERENCES Produtos(id_produto)
);
